void sortZeroesAndOne(int *arr, int n)
{
    int left = 0, right = n-1;

    while(left < right) {
        while(arr[left] == 0) {
            left++;
        }

        while(arr[right] == 1) {
            right--;
        }
        if(arr[left]==1 && arr[right]==0 && left<right) {
        arr[left] = 0;
        arr[right] = 1;
        left++;
        right--;
        }
    }
    

}
